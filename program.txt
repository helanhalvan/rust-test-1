add:
    (A, B) /* variable bind */
    { (A =/= 0) && (B =/= 0) } /* precondition expr */
    / A + B \ /* evaluation */
    { A == 0 }
    / B \
    { B == 0 }
    / A \
end

add1:
    (A)
    / add(A, 1) \
end

map:
    ([], F, A)
    / A \
    ([H|T], F, Acc)
    / map(T, F, [Acc|F(H)]) \
end

main: 
    ()
    / map([5, 3], add1, []) \ 
end
