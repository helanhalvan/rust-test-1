add:
    (A, B) /* variable bind */
    { (A =/= 0) && (B =/= 0) } /* precondition expr */
    / A + B \ /* evaluation */
    { A == 0 }
    / B \
    { B == 0 }
    / A \
end

add1:
    ([], F, A)
    / A \
    ([H|T], F, A)
    / add1(T, F, [A|add(H,1)]) \
end

main: 
    ()
    / add1([5, 3], 1, []) \ 
end
